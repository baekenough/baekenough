#!/bin/bash # checkpoint.sh - with circuit breaker MAX_RETRIES=3 RETRY=0 while [ $RETRY -lt $MAX_RETRIES ]; do OUTPUT=$(claude -p "Create progress summary checkpoint" --allowedTools "Read,Glob,Grep" 2>&1) if echo "$OUTPUT" | grep -qi "rate.limit\|usage.limit"; then echo "Rate limited, waiting 60s (attempt $((RETRY+1))/$MAX_RETRIES)" sleep 60 RETRY=$((RETRY+1)) else echo "$OUTPUT" exit 0 fi done echo "ERROR: Rate limit not resolved after $MAX_RETRIES attempts" && exit 1
